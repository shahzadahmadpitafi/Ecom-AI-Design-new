## STEP 5 — API ROUTE (/app/api/edit-design/route.ts)

```typescript
import { GoogleGenerativeAI } from '@google/generative-ai';
import { NextRequest, NextResponse } from 'next/server';

export async function POST(req: NextRequest) {
  const body = await req.json();
  const { currentImageBase64, currentImageMimeType, editPrompt, logoBase64, logoMimeType } = body;
  const apiKey = req.headers.get('x-gemini-key') || process.env.GEMINI_API_KEY!;
  
  const genAI = new GoogleGenerativeAI(apiKey);
  const model = genAI.getGenerativeModel({ model: 'gemini-2.0-flash-exp' });

  // Build content parts
  const parts: any[] = [
    {
      inlineData: {
        mimeType: currentImageMimeType || 'image/png',
        data: currentImageBase64,
      }
    }
  ];

  // If logo is provided, add it as second image
  if (logoBase64) {
    parts.push({
      inlineData: {
        mimeType: logoMimeType || 'image/png',
        data: logoBase64,
      }
    });
  }

  // Add the text instruction last
  parts.push({ text: editPrompt });

  const result = await model.generateContent({
    contents: [{ role: 'user', parts }],
    generationConfig: { responseModalities: ['image', 'text'] },
  });

  const responseParts = result.response.candidates?.[0]?.content?.parts;
  const imagePart = responseParts?.find(p => p.inlineData);
  
  if (imagePart?.inlineData) {
    return NextResponse.json({
      success: true,
      imageUrl: `data:${imagePart.inlineData.mimeType};base64,${imagePart.inlineData.data}`
    });
  }

  return NextResponse.json({ error: 'Edit failed' }, { status: 500 });
}
```

```typescript
const handleApplyEdit = async () => {
  // Convert current canvas image to base64
  const currentImg = document.querySelector('.canvas-area img') as HTMLImageElement;
  const canvas = document.createElement('canvas');
  canvas.width = currentImg.naturalWidth;
  canvas.height = currentImg.naturalHeight;
  const ctx = canvas.getContext('2d')!;
  ctx.drawImage(currentImg, 0, 0);
  const currentImageBase64 = canvas.toDataURL('image/png').split(',')[1];

  setCanvasMode('generating'); // show loading state
  setLoadingMessage('Applying your edit...');

  const response = await fetch('/api/edit-design', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'x-gemini-key': localStorage.getItem('gemini_api_key') || '',
    },
    body: JSON.stringify({
      currentImageBase64,
      currentImageMimeType: 'image/png',
      editPrompt: buildEditPrompt(), // builds prompt from current selections
      logoBase64: uploadedLogoBase64 || null,
      logoMimeType: 'image/png',
    }),
  });

  const data = await response.json();
  if (data.imageUrl) {
    // Add original to history before replacing
    setDesignHistory(prev => [{ imageUrl: generatedImageUrl, prompt: 'Before edit', timestamp: new Date() }, ...prev].slice(0, 10));
    setGeneratedImageUrl(data.imageUrl);
    setCanvasMode('generated');
    toast.success('Edit applied!');
  }
};
```
