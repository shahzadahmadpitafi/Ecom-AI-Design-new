## STEP 2 — HANDLE NATURAL LANGUAGE EDIT

```typescript
const handleNaturalLanguageEdit = async (editInstruction: string) => {
  if (!generatedImageUrl && !viewCache[currentView]) {
    toast.error('Generate a design first before editing');
    return;
  }

  setIsApplyingEdit(true);
  setEditLoadingMessage(editInstruction); // shows what's being applied

  // Get current image as base64
  const currentImageUrl = viewCache[currentView] || generatedImageUrl;
  let currentImageBase64 = '';
  let currentImageMimeType = 'image/png';

  try {
    // Convert current image to base64 for Gemini
    if (currentImageUrl.startsWith('data:')) {
      // Already base64
      const parts = currentImageUrl.split(',');
      currentImageMimeType = parts[0].split(':')[1].split(';')[0];
      currentImageBase64 = parts[1];
    } else {
      // URL — fetch and convert
      const response = await fetch('/api/fetch-image-base64', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ imageUrl: currentImageUrl }),
      });
      const data = await response.json();
      currentImageBase64 = data.base64;
      currentImageMimeType = data.mimeType;
    }

    // Build the edit prompt
    const editPrompt = buildNaturalLanguageEditPrompt(editInstruction);

    // Call edit API
    const response = await fetch('/api/edit-design', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-gemini-key': localStorage.getItem('gemini_api_key') || '',
      },
      body: JSON.stringify({
        currentImageBase64,
        currentImageMimeType,
        editPrompt,
        editType: 'natural_language',
        originalInstruction: editInstruction,
      }),
    });

    const data = await response.json();

    if (data.imageUrl) {
      // Save to history before replacing
      setDesignHistory(prev => [{
        imageUrl: viewCache[currentView] || generatedImageUrl,
        prompt: `Before: "${editInstruction}"`,
        timestamp: new Date(),
        type: 'pre-edit',
      }, ...prev].slice(0, 10));

      // Update view cache and canvas
      setViewCache(prev => ({ ...prev, [currentView]: data.imageUrl }));
      setGeneratedImageUrl(data.imageUrl);
      
      // Add edited version to history
      setDesignHistory(prev => [{
        imageUrl: data.imageUrl,
        prompt: `Edited: "${editInstruction}"`,
        timestamp: new Date(),
        type: 'edited',
      }, ...prev].slice(0, 10));

      toast.success(`Applied: "${editInstruction}"`, {
        style: { background: '#1a1a1a', color: '#C9A84C', border: '1px solid rgba(201,168,76,0.3)' }
      });
    }
  } catch (error) {
    toast.error('Edit failed — try rephrasing your instruction');
  } finally {
    setIsApplyingEdit(false);
    setEditLoadingMessage('');
  }
};
```

- The result must look photorealistic — like a real manufactured garment

- Professional product photography lighting
- High resolution, realistic fabric texture

Apply the change naturally as if the garment was originally manufactured this way.
Do not add anything the user didn't ask for.
Do not remove anything the user didn't ask to remove.`;

## STEP 7 — /api/fetch-image-base64 ROUTE

Create this helper route for converting URL images to base64:

```typescript
// /app/api/fetch-image-base64/route.ts
export async function POST(req: NextRequest) {
  const { imageUrl } = await req.json();

  const response = await fetch(imageUrl);
  const buffer = await response.arrayBuffer();
  const base64 = Buffer.from(buffer).toString('base64');
  const mimeType = response.headers.get('content-type') || 'image/png';

  return NextResponse.json({ base64, mimeType });
}
```
