## THE EXACT BEHAVIOR REQUIRED

When a design is generated and showing on canvas:
- User clicks "BACK" → Gemini regenerates the SAME garment from the BACK view → canvas updates
- User clicks "LEFT SLEEVE" → Gemini regenerates showing the LEFT SLEEVE view → canvas updates
- User clicks "RIGHT SLEEVE" → Gemini regenerates showing the RIGHT SLEEVE view → canvas updates
- User clicks "FRONT" → returns to front view (either regenerates or restores cached front image)

Each view should look like a professional product photo from that specific angle.

## STEP 2 — VIEW SWITCH HANDLER

```typescript
const handleViewChange = async (newView: 'front' | 'back' | 'left sleeve' | 'right sleeve') => {
  // Don't do anything if already on this view
  if (newView === currentView) return;
  
  // Don't switch if no design has been generated yet
  if (canvasMode === 'empty') {
    setCurrentView(newView);
    return;
  }

  // Check cache first — if we already generated this view, restore it instantly
  if (viewCache[newView]) {
    setCurrentView(newView);
    setGeneratedImageUrl(viewCache[newView]);
    return; // no API call needed
  }

  // Need to generate this view
  setCurrentView(newView);
  setIsViewSwitching(true);
  setCanvasMode('generating');

  try {
    const response = await fetch('/api/generate-design', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-gemini-key': localStorage.getItem('gemini_api_key') || '',
      },
      body: JSON.stringify({
        prompt: currentPrompt, // the original prompt used to generate
        styleModifiers: currentStyleModifiers,
        garmentType: selectedProduct || 'T-Shirt',
        garmentColor: selectedColor || 'Black',
        view: newView, // ← pass the new view
      }),
    });

    const data = await response.json();

    if (data.imageUrl) {
      // Cache this view
      setViewCache(prev => ({ ...prev, [newView]: data.imageUrl }));
      setGeneratedImageUrl(data.imageUrl);
      setCanvasMode('generated');
    }
  } catch (error) {
    // On error, revert to previous view
    setCurrentView(currentView);
    setCanvasMode('generated');
    toast.error('Could not load this view — please try again');
  } finally {
    setIsViewSwitching(false);
  }
};
```

## STEP 5 — VIEW TOGGLE BUTTON UI

Update the view toggle buttons to show correct states:

```typescript
const views = ['front', 'back', 'left sleeve', 'right sleeve'] as const;

// Button styling logic:
const getViewButtonStyle = (view: string) => {
  const isActive = currentView === view;
  const isCached = !!viewCache[view]; // already generated, will load instantly
  const isLoading = isViewSwitching && currentView === view;
  
  return {
    background: isActive ? '#C9A84C' : 'transparent',
    color: isActive ? '#000' : '#a0a0a0',
    border: isActive 
      ? 'none' 
      : isCached 
        ? '1px solid rgba(201,168,76,0.4)' // gold border = cached/ready
        : '1px solid rgba(167,139,250,0.3)', // purple border = not yet generated
    cursor: isLoading ? 'wait' : 'pointer',
    opacity: isLoading ? 0.7 : 1,
    position: 'relative' as const,
  };
};
```

Button JSX:
```tsx
<div style={{ display: 'flex', gap: '4px' }}>
  {views.map((view) => {
    const isActive = currentView === view;
    const isCached = !!viewCache[view];
    const isThisLoading = isViewSwitching && currentView === view;
    
    return (
      <button
        key={view}
        onClick={() => handleViewChange(view)}
        disabled={isViewSwitching}
        style={getViewButtonStyle(view)}
      >
        {isThisLoading ? (
          // Show spinner inside button while this view is loading
          <span style={{ display: 'flex', alignItems: 'center', gap: '4px' }}>
            <SpinnerIcon size={12} />
            ...
          </span>
        ) : (
          <>
            {view.toUpperCase()}
            {isCached && !isActive && (
              // Small gold dot indicates cached/instant load
              <span style={{
                width: 4, height: 4,
                background: '#C9A84C',
                borderRadius: '50%',
                position: 'absolute',
                top: 4, right: 4,
              }} />
            )}
          </>
        )}
      </button>
    );
  })}
</div>
```

## STEP 9 — PREFETCH BACK VIEW

After front view generates successfully, silently prefetch the back view in background:

```typescript
const prefetchBackView = async () => {
  if (viewCache['back']) return; // already cached
  
  try {
    const response = await fetch('/api/generate-design', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'x-gemini-key': localStorage.getItem('gemini_api_key') || '' },
      body: JSON.stringify({
        prompt: currentPrompt,
        styleModifiers: currentStyleModifiers,
        garmentType: selectedProduct,
        garmentColor: selectedColor,
        view: 'back',
      }),
    });
    const data = await response.json();
    if (data.imageUrl) {
      setViewCache(prev => ({ ...prev, back: data.imageUrl }));
      // The BACK button will now show the gold dot (cached indicator)
    }
  } catch {
    // Silent fail — prefetch is optional
  }
};

// Call after successful generation (with 2s delay):
setTimeout(prefetchBackView, 2000);
```
